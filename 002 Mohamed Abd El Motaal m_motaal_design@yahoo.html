def rem_outliers(df_in):
    df=df_in
    for col_lable in df.columns.tolist():
        if (col_lable=='state') or (col_lable=='month_x') or  (col_lable=='month_y'):
            continue
        print(col_lable)
        q1=df[col_lable].quantile(0.25)
        q3=df[col_lable].quantile(0.75)
        IQR=q3-q1
        outlaiers_lower_R=q1-1.5*IQR
        outlaiers_Upper_R=q3+1.5*IQR
        print('column :{}  {}  to   {}'.format(col_lable,outlaiers_lower_R, outlaiers_Upper_R))
        df1=df
        df1_outliers_smaller_indx=df1[df1[col_lable] < outlaiers_lower_R].index.tolist()

        print('df1_outliers_smaller_indx: {},  df1_outliers_greater_indx:  {}  '.format(df1_outliers_smaller_indx,df1_outliers_greater_indx))
        df1.drop(index=df1[df1[col_lable] < outlaiers_lower_R].index.tolist(),axis=1,inplace=True)
        df1.drop(index=df1[df1[col_lable] > outlaiers_Upper_R].index.tolist(),axis=1,inplace=True)
        lis=df1[col_lable].values.tolist()

def gun_per_cap(data_frame):  #defining afunction that recieves a data frame and returns gun per capita ratio.
    pop10=[]
    pop16=[]
    tot10=[]
    tot16=[]
    for i in range(0,len(data_frame.iloc[:,0])):
        pop10.append(totals.iloc[i,2].sum())
        pop16.append(totals.iloc[i,5].sum())
        tot10.append(totals.iloc[i,3].sum())
        tot16.append(totals.iloc[i,6].sum())
    spop10=sum(pop10)
    spop16=sum(pop16)
    stot10=sum(tot10)
    stot16=sum(tot16)
    gpc10=stot10/spop10
    gpc16=stot16/spop16
    gpc=[gpc10,gpc16]
    return gpc #print('GunPerCapita ratio for both yers, 2010 and 2016 are: {} and {}.'.format(gpc[0],gpc[1]))
